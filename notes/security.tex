\section{Secret-key cryptography}
\begin{itemize}
    \item monoalphabetic substitution cipher: frequency attack

    \item kerchoff's principle: same algorithm, different keys

    \item public key cryptography: anyone can encrypt for you, only you can decrypt

    \item digital signatures: decrypt(hash), then encrypt(decrypt(hash)) = hash to check tag

    \item public key infra: browser comes with certificate authority public key, can check
\end{itemize}


\section{Access Control}
\begin{itemize}
    \item groups with permissions
    \item protection matrix: most are empty, inefficient
    \item access control list: each file has a linked list of users w/ permissions
    \item capabilty list: each user has list of files and permissions
\end{itemize}


\section{Bell-La Padula}
\begin{itemize}
    \item military
    \item simple security property: process can only read objects lower than security level

        security level 3 reads from 1..3

    \item * property: process can only write at level or higher

        soldiers pass info to general 
\end{itemize}


\section{Biba model}
\begin{itemize}
    \item corporate
    \item simple integrity property: can write lower

        boss tells workers what do do

    \item integrity * property: can read higher 

        workers read boss' instructions
\end{itemize}
